function mygrep { grep -rnIi "$1" . --color; }
alias glp="git log --graph --pretty=format:'%Cred%h%Creset -%C(yellow)%d%Creset %s %Cblueby %an %Cgreen(%cr)%Creset'"

    # 自動載入 Alias
